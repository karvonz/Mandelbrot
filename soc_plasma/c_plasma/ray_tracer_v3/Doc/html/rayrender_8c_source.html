78  // convert pixels to image plane coordinates (focal length = 1.0)
79  vec3_t ray = MathVec3Set( sub( ( div( x, (number)size ) ), T( 0.5f ) ), - ( sub( ( div( y, (number)size ) ), T( 0.5f ) ) ), T( 1.0f ) );
80  rayintersect_t info;
81  rayintersect_t * infoptr = &info;
82  // raytrace - intersected objects are stored in a pointer
83  if ( Raytrace( ray, g_origin, infoptr ) ) {
84  // intersection
85  // 3D point of intersection
86  g_point = MathVec3Scale( ray, infoptr->dist );
87  // is there a sphere with a mirror surface ?
88  if ( infoptr->type == OBJECT_TYPE_SPHERE && SphereHasReflection( infoptr->index ) ) {
89  // reflect ray off the surface
90  ray = Reflect( infoptr->type, infoptr->index, ray, g_origin );
91  // follow the reflected ray
92  if ( Raytrace( ray, g_point, infoptr ) ) {
93  // 3D point of intersection
94  g_point = MathVec3Add( MathVec3Scale( ray, infoptr->dist ), g_point );
95  }
96  }

99  int type = infoptr->type;
100  int index = infoptr->index;

102  number i = g_ambient;
