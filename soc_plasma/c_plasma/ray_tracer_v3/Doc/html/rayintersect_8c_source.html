23  vec3_t s = MathVec3Sub( SphereGetPosition( idx ), o ); // sphere center translated into coordinate frame of ray origin
24  number radius = SphereGetRadius( idx ); // sphere radius
25  number a = MathVec3Dot( r, r ); // intersection of sphere and line = quadratic function of distance
26  number b = mul( T( -2.0f ), MathVec3Dot( s, r ) ); // a x^2 + b x + c = 0
27  number c = sub( MathVec3Dot( s, s ), ( mul( radius, radius ) ) ); // (r'r)x^2 - (2s'r)x + (s's - radius^2) = 0
28  number d = sub( mul( b, b ), mul( T( 4.0f ), mul( a, c ) ) ); // precompute discriminant
29  if ( d > T( 0.0f ) ) { // solution exists only if sqrt( d ) is real (not imaginary)

32  if ( IntersectionTest( dist, OBJECT_TYPE_SPHERE, idx, info ) ) { // is this closest intersection so far ?
